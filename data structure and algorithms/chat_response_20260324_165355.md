class DoublyLinkedList {
    // Node class representing each element in the list
    class Node {
        int data;
        Node prev;
        Node next;

        Node(int data) {
            this.data = data;
        }
    }

    private Node head;
    private Node tail;

    // Method to add a node at the end of the list
    public void append(int data) {
        Node newNode = new Node(data);
        if (head == null) {
            head = tail = newNode;
        } else {
            tail.next = newNode;
            newNode.prev = tail;
            tail = newNode;
        }
    }

    // Method to add a node at the beginning of the list
    public void prepend(int data) {
        Node newNode = new Node(data);
        if (head == null) {
            head = tail = newNode;
        } else {
            newNode.next = head;
            head.prev = newNode;
            head = newNode;
        }
    }

    // Method to delete a node from the list
    public void delete(int data) {
        Node current = head;
        while (current != null) {
            if (current.data == data) {
                if (current.prev != null) {
                    current.prev.next = current.next;
                } else {
                    head = current.next;
                }
                if (current.next != null) {
                    current.next.prev = current.prev;
                } else {
                    tail = current.prev;
                }
                return;
            }
            current = current.next;
        }
    }

    // Method to display the list from head to tail
    public void displayForward() {
        Node current = head;
        while (current != null) {
            System.out.print(current.data + " ");
            current = current.next;
        }
        System.out.println();
    }

    // Method to display the list from tail to head
    public void displayBackward() {
        Node current = tail;
        while (current != null) {
            System.out.print(current.data + " ");
            current = current.prev;
        }
        System.out.println();
    }

    public static void main(String[] args) {
        DoublyLinkedList dll = new DoublyLinkedList();

        // Append nodes
        dll.append(10);
        dll.append(20);
        dll.append(30);

        // Prepend nodes
        dll.prepend(5);
        dll.prepend(1);

        // Display list forward
        System.out.println("List displayed forward:");
        dll.displayForward();

        // Display list backward
        System.out.println("List displayed backward:");
        dll.displayBackward();

        // Delete a node
        dll.delete(20);

        // Display list after deletion
        System.out.println("List after deleting 20:");
        dll.displayForward();
    }
}

### Explanation:
- **Node Class**: Represents each element in the list with `data`, `prev`, and `next` pointers.
- **append(int data)**: Adds a new node at the end of the list.

- **delete(int data)**: Removes the first occurrence of a node with the specified data.
- **displayForward()**: Prints the list from head to tail.
- **displayBackward()**: Prints the list from tail to head.
- **main()**: Demonstrates the usage of the doubly linked list by appending, prepending, deleting, and displaying nodes.
