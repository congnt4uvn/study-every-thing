Dưới đây là một ví dụ về mã Java cho thuật toán mã hóa Huffman, bao gồm các cấu trúc dữ liệu và hàm chính để nén dữ liệu. Mã này sẽ tạo mã tiền tố tối ưu cho việc nén dữ liệu.

```java
import java.util.PriorityQueue;
import java.util.Comparator;
import java.util.HashMap;
import java.util.Map;

// Node class đại diện cho một nút trong cây Huffman
class Node {
    char ch;
    int frequency;
    Node left = null, right = null;

    Node(char ch, int frequency) {
        this.ch = ch;
        this.frequency = frequency;
    }

    public Node(char ch, int frequency, Node left, Node right) {
        this.ch = ch;
        this.frequency = frequency;
        this.left = left;
        this.right = right;
    }
}

// Comparator để sắp xếp các nút theo tần số
class HuffmanComparator implements Comparator<Node> {
    public int compare(Node left, Node right) {
        return left.frequency - right.frequency;
    }
}

public class HuffmanCoding {

    // Hàm để xây dựng cây Huffman và trả về nút gốc
    public static Node buildHuffmanTree(Map<Character, Integer> freq) {
        PriorityQueue<Node> pq = new PriorityQueue<>(new HuffmanComparator());

        // Tạo một nút lá cho mỗi ký tự và thêm vào hàng đợi ưu tiên
        for (var entry : freq.entrySet()) {
            pq.add(new Node(entry.getKey(), entry.getValue()));
        }

        // Lặp lại cho đến khi hàng đợi chỉ còn một nút
        while (pq.size() != 1) {
            // Xóa hai nút có tần số thấp nhất
            Node left = pq.poll();
            Node right = pq.poll();

            // Tạo một nút nội bộ với hai nút con
            int sum = left.frequency + right.frequency;
            pq.add(new Node('\0', sum, left, right));
        }

        // Nút gốc là nút duy nhất còn lại trong hàng đợi
        return pq.peek();
    }

    // Hàm để duyệt cây Huffman và lưu trữ mã Huffman trong một map
    public static void encode(Node root, String str, Map<Character, String> huffmanCode) {
        if (root == null) {
            return;
        }

        // Tìm thấy một nút lá
        if (root.left == null && root.right == null) {
            huffmanCode.put(root.ch, str);
        }

        encode(root.left, str + "0", huffmanCode);
        encode(root.right, str + "1", huffmanCode);
    }

    // Hàm giải mã một chuỗi mã hóa
    public static String decode(Node root, String encodedString) {
        StringBuilder sb = new StringBuilder();
        Node current = root;
        for (int i = 0; i < encodedString.length(); i++) {
            current = (encodedString.charAt(i) == '0') ? current.left : current.right;

            // Tìm thấy một nút lá
            if (current.left == null && current.right == null) {
                sb.append(current.ch);
                current = root;
            }
        }
        return sb.toString();
    }

    public static void main(String[] args) {
        String text = "Huffman coding is a data compression algorithm.";

        // Đếm tần số xuất hiện của mỗi ký tự
        Map<Character, Integer> freq = new HashMap<>();
        for (char c : text.toCharArray()) {
            freq.put(c, freq.getOrDefault(c, 0) + 1);
        }

        // Xây dựng cây Huffman
        Node root = buildHuffmanTree(freq);

        // Lưu trữ mã Huffman trong một map
        Map<Character, String> huffmanCode = new HashMap<>();
        encode(root, "", huffmanCode);

        // In mã Huffman
        System.out.println("Huffman Codes are: " + huffmanCode);
        System.out.println("Original string: " + text);

        // Mã hóa chuỗi
        StringBuilder encodedString = new StringBuilder();
        for (char c : text.toCharArray()) {
            encodedString.append(huffmanCode.get(c));
        }

        System.out.println("Encoded string: " + encodedString);

        // Giải mã chuỗi
        String decodedString = decode(root, encodedString.toString());
        System.out.println("Decoded string: " + decodedString);
    }
}
```

### Giải thích:
- **Node**: Lớp đại diện cho một nút trong cây Huffman.
- **HuffmanComparator**: Comparator để sắp xếp các nút theo tần số.
- **buildHuffmanTree**: Hàm để xây dựng cây Huffman từ tần số ký tự.
- **encode**: Hàm để tạo mã Huffman cho mỗi ký tự.
- **decode**: Hàm để giải mã một chuỗi mã hóa.
- **main**: Hàm chính để chạy chương trình, bao gồm việc mã hóa và giải mã một chuỗi văn bản.
