Dưới đây là một ví dụ về cách triển khai cây đỏ-đen (Red-Black Tree) bằng Java, bao gồm các chức năng chính của cấu trúc dữ liệu và thuật toán. Mã này bao gồm các phương thức để chèn, xóa và duyệt cây. Tôi sẽ cung cấp các chú thích bằng tiếng Việt để giúp bạn hiểu rõ hơn.

```java
// Lớp Node đại diện cho một nút trong cây đỏ-đen
class Node {
    int data;
    Node parent;
    Node left;
    Node right;
    int color; // 0 cho đen, 1 cho đỏ

    public Node(int data) {
        this.data = data;
        this.color = 1; // Mặc định là đỏ khi chèn
    }
}

// Lớp RedBlackTree chứa các phương thức chính
public class RedBlackTree {
    private Node root;
    private Node TNULL;

    // Khởi tạo cây với nút TNULL
    public RedBlackTree() {
        TNULL = new Node(0);
        TNULL.color = 0;
        TNULL.left = null;
        TNULL.right = null;
        root = TNULL;
    }

    // Tiện ích để thực hiện phép quay trái
    private void leftRotate(Node x) {
        Node y = x.right;
        x.right = y.left;
        if (y.left != TNULL) {
            y.left.parent = x;
        }
        y.parent = x.parent;
        if (x.parent == null) {
            this.root = y;
        } else if (x == x.parent.left) {
            x.parent.left = y;
        } else {
            x.parent.right = y;
        }
        y.left = x;
        x.parent = y;
    }

    // Tiện ích để thực hiện phép quay phải
    private void rightRotate(Node x) {
        Node y = x.left;
        x.left = y.right;
        if (y.right != TNULL) {
            y.right.parent = x;
        }
        y.parent = x.parent;
        if (x.parent == null) {
            this.root = y;
        } else if (x == x.parent.right) {
            x.parent.right = y;
        } else {
            x.parent.left = y;
        }
        y.right = x;
        x.parent = y;
    }

    // Sửa cây sau khi chèn
    private void fixInsert(Node k) {
        Node u;
        while (k.parent.color == 1) {
            if (k.parent == k.parent.parent.right) {
                u = k.parent.parent.left;
                if (u.color == 1) {
                    u.color = 0;
                    k.parent.color = 0;
                    k.parent.parent.color = 1;
                    k = k.parent.parent;
                } else {
                    if (k == k.parent.left) {
                        k = k.parent;
                        rightRotate(k);
                    }
                    k.parent.color = 0;
                    k.parent.parent.color = 1;
                    leftRotate(k.parent.parent);
                }
            } else {
                u = k.parent.parent.right;
                if (u.color == 1) {
                    u.color = 0;
                    k.parent.color = 0;
                    k.parent.parent.color = 1;
                    k = k.parent.parent;
                } else {
                    if (k == k.parent.right) {
                        k = k.parent;
                        leftRotate(k);
                    }
                    k.parent.color = 0;
                    k.parent.parent.color = 1;
                    rightRotate(k.parent.parent);
                }
            }
            if (k == root) {
                break;
            }
        }
        root.color = 0;
    }

    // Chèn một nút vào cây
    public void insert(int key) {
        Node node = new Node(key);
        node.parent = null;
        node.data = key;
        node.left = TNULL;
        node.right = TNULL;
        node.color = 1;

        Node y = null;
        Node x = this.root;

        while (x != TNULL) {
            y = x;
            if (node.data < x.data) {
                x = x.left;
            } else {
                x = x.right;
            }
        }

        node.parent = y;
        if (y == null) {
            root = node;
        } else if (node.data < y.data) {
            y.left = node;
        } else {
            y.right = node;
        }

        if (node.parent == null) {
            node.color = 0;
            return;
        }

        if (node.parent.parent == null) {
            return;
        }

        fixInsert(node);
    }

    // In cây theo thứ tự
    public void inorder() {
        inorderHelper(this.root);
    }

    private void inorderHelper(Node node) {
        if (node != TNULL) {
            inorderHelper(node.left);
            System.out.print(node.data + " ");
            inorderHelper(node.right);
        }
    }

    // Hàm chính để chạy chương trình
    public static void main(String[] args) {
        RedBlackTree tree = new RedBlackTree();
        
        // Chèn các phần tử vào cây
        tree.insert(20);
        tree.insert(15);
        tree.insert(25);
        tree.insert(10);
        tree.insert(5);
        tree.insert(1);

        // In cây theo thứ tự
        System.out.println("Inorder traversal:");
        tree.inorder();
    }
}
```

### Giải thích:
- **Node**: Lớp này đại diện cho một nút trong cây đỏ-đen, với các thuộc tính như `data`, `parent`, `left`, `right`, và `color`.
- **RedBlackTree**: Lớp này chứa các phương thức để thực hiện các thao tác trên cây như chèn, quay trái, quay phải, và sửa cây sau khi chèn.
- **insert**: Phương thức này chèn một nút mới vào cây và sau đó gọi `fixInsert` để đảm bảo các thuộc tính của cây đỏ-đen được duy trì.
- **inorder**: Phương thức này in các phần tử của cây theo thứ tự tăng dần.

Bạn có thể chạy chương trình này để thấy cách cây đỏ-đen hoạt động với các thao tác chèn và duyệt cây.
