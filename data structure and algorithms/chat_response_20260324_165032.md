Dưới đây là một ví dụ về cách triển khai cấu trúc dữ liệu Heap trong Java, bao gồm cả Min Heap và Max Heap. Mã này bao gồm các chức năng chính như thêm phần tử, xóa phần tử nhỏ nhất/lớn nhất và hiển thị heap. 

```java
import java.util.Arrays;

class MinHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MinHeap(int capacity) {
        this.capacity = capacity;
        this.size = 0;
        this.heap = new int[capacity];
    }

    private int parent(int index) { return (index - 1) / 2; }
    private int leftChild(int index) { return 2 * index + 1; }
    private int rightChild(int index) { return 2 * index + 2; }

    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    public void insert(int key) {
        if (size == capacity) {
            System.out.println("Heap đã đầy");
            return;
        }
        heap[size] = key;
        int current = size;
        size++;

        while (current != 0 && heap[parent(current)] > heap[current]) {
            swap(current, parent(current));
            current = parent(current);
        }
    }

    public int extractMin() {
        if (size <= 0) return Integer.MAX_VALUE;
        if (size == 1) {
            size--;
            return heap[0];
        }

        int root = heap[0];
        heap[0] = heap[size - 1];
        size--;
        minHeapify(0);

        return root;
    }

    private void minHeapify(int index) {
        int smallest = index;
        int left = leftChild(index);
        int right = rightChild(index);

        if (left < size && heap[left] < heap[smallest]) smallest = left;
        if (right < size && heap[right] < heap[smallest]) smallest = right;

        if (smallest != index) {
            swap(index, smallest);
            minHeapify(smallest);
        }
    }

    public void printHeap() {
        System.out.println("Min Heap: " + Arrays.toString(Arrays.copyOfRange(heap, 0, size)));
    }
}

class MaxHeap {
    private int[] heap;
    private int size;
    private int capacity;

    public MaxHeap(int capacity) {
        this.capacity = capacity;
        this.size = 0;
        this.heap = new int[capacity];
    }

    private int parent(int index) { return (index - 1) / 2; }
    private int leftChild(int index) { return 2 * index + 1; }
    private int rightChild(int index) { return 2 * index + 2; }

    private void swap(int i, int j) {
        int temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    public void insert(int key) {
        if (size == capacity) {
            System.out.println("Heap đã đầy");
            return;
        }
        heap[size] = key;
        int current = size;
        size++;

        while (current != 0 && heap[parent(current)] < heap[current]) {
            swap(current, parent(current));
            current = parent(current);
        }
    }

    public int extractMax() {
        if (size <= 0) return Integer.MIN_VALUE;
        if (size == 1) {
            size--;
            return heap[0];
        }

        int root = heap[0];
        heap[0] = heap[size - 1];
        size--;
        maxHeapify(0);

        return root;
    }

    private void maxHeapify(int index) {
        int largest = index;
        int left = leftChild(index);
        int right = rightChild(index);

        if (left < size && heap[left] > heap[largest]) largest = left;
        if (right < size && heap[right] > heap[largest]) largest = right;

        if (largest != index) {
            swap(index, largest);
            maxHeapify(largest);
        }
    }

    public void printHeap() {
        System.out.println("Max Heap: " + Arrays.toString(Arrays.copyOfRange(heap, 0, size)));
    }
}

public class HeapExample {
    public static void main(String[] args) {
        MinHeap minHeap = new MinHeap(10);
        minHeap.insert(3);
        minHeap.insert(2);
        minHeap.insert(15);
        minHeap.printHeap();
        System.out.println("Phần tử nhỏ nhất: " + minHeap.extractMin());
        minHeap.printHeap();

        MaxHeap maxHeap = new MaxHeap(10);
        maxHeap.insert(3);
        maxHeap.insert(2);
        maxHeap.insert(15);
        maxHeap.printHeap();
        System.out.println("Phần tử lớn nhất: " + maxHeap.extractMax());
        maxHeap.printHeap();
    }
}
```

### Giải thích:
- **MinHeap**: Cấu trúc dữ liệu này duy trì phần tử nhỏ nhất ở gốc. Các phương thức chính bao gồm `insert`, `extractMin`, và `minHeapify`.
- **MaxHeap**: Cấu trúc dữ liệu này duy trì phần tử lớn nhất ở gốc. Các phương thức chính bao gồm `insert`, `extractMax`, và `maxHeapify`.
- **HeapExample**: Lớp này chứa phương thức `main` để minh họa cách sử dụng Min Heap và Max Heap.

Chạy chương trình này sẽ cho bạn thấy cách các phần tử được thêm vào và trích xuất từ Min Heap và Max Heap.
